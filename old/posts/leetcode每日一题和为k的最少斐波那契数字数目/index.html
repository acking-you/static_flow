class Solution {
public:
    int findMinFibonacciNumbers(int k) {
        vector<int>items(2,1);
        int ret = 0;
        while(items.back()<k){
            items.push_back(items.back()+items[items.size()-2]);
        }
        for(int i=items.size()-1;i>=0;i--){
            ret += k/items[i];
            k %= items[i];
        }
        return ret;
    }
};
